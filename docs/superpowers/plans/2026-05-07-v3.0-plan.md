# R+ 3.0.0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement R+ 3.0.0: introduce a per-device "modo de trabajo" (Sala / Interconsulta) that reshapes the patient expediente, replacing Nota de Evolución with a quick "Estado Actual" capture and adding a new "Listado de Problemas" tab with .docx generation; add Fecha de Nacimiento (DOB) as required for new patients with age calculated.

**Architecture:** Pragmatic `if (isModeSala())` checks at call sites (no abstraction layer). DOB stored as `'YYYY-MM-DD'`, edad string still cached for backward compatibility. Estado Actual reuses the existing `#soap-modal-backdrop` with conditional rendering via `body[data-estado-actual-mode]` CSS. Listado de Problemas mirrors the Indicaciones tab pattern. New Python script `generate_listado.py` follows the proven pattern of `generate_note.py`/`generate_indicaciones.py`. Migration is non-blocking and idempotent.

**Tech Stack:** Electron 41, vanilla JS (ES modules), Express 5, Python 3 stdlib (DOCX manipulation), Node `node --test` for unit tests.

**Spec:** [`docs/superpowers/specs/2026-05-07-v3.0-design.md`](../specs/2026-05-07-v3.0-design.md)

```js
// Modo de trabajo del usuario y migración de settings v3.0.

export function isModeSala(settings) {
  if (!settings) return true;
  return (settings.appMode || 'sala') === 'sala';
}

export function getDefaultServicio(settings) {
  if (!settings) return '';
  return String(settings.defaultServicio || '').trim();
}

/**
 * Migración suave a 3.0.0. Idempotente.
 * Muta el settings recibido y retorna true si aplicó la migración, false si ya estaba migrado.
 */
export function migrateToV3(settings) {
  if (!settings || settings._v3MigrationDone) return false;
  if (settings.appMode == null) settings.appMode = 'sala';
  if (settings.defaultServicio == null) settings.defaultServicio = '';
  settings._v3MigrationDone = true;
  return true;
}
```

```js
// Edad calculada desde fecha de nacimiento. Usa UTC para evitar drift de timezone.

const DOB_RE = /^(\d{4})-(\d{2})-(\d{2})$/;

function parseDobToUTC(dob) {
  if (typeof dob !== 'string') return null;
  const m = DOB_RE.exec(dob);
  if (!m) return null;
  const y = Number(m[1]), mo = Number(m[2]), d = Number(m[3]);
  if (mo < 1 || mo > 12 || d < 1 || d > 31) return null;
  const ts = Date.UTC(y, mo - 1, d);
  const date = new Date(ts);
  // sanity: re-derive Y/M/D from UTC (rejects 02-30, etc.)
  if (date.getUTCFullYear() !== y || date.getUTCMonth() !== mo - 1 || date.getUTCDate() !== d) return null;
  return ts;
}

function asOfToUTCDay(asOf) {
  const d = asOf instanceof Date ? asOf : new Date(asOf);
  if (isNaN(d.getTime())) return null;
  return Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate());
}

export function calculateAge(dob, asOf = new Date()) {
  const dobMs = parseDobToUTC(dob);
  const todayMs = asOfToUTCDay(asOf);
  if (dobMs == null || todayMs == null) return null;
  if (dobMs > todayMs) return null;

  const dy = new Date(dobMs).getUTCFullYear();
  const ty = new Date(todayMs).getUTCFullYear();
  if (ty - dy > 120) return null;

  // años completos
  let years = ty - dy;
  const dmd = (new Date(dobMs).getUTCMonth() * 100) + new Date(dobMs).getUTCDate();
  const tmd = (new Date(todayMs).getUTCMonth() * 100) + new Date(todayMs).getUTCDate();
  if (tmd < dmd) years -= 1;

  if (years >= 1) return { value: years, unit: 'años', display: years + ' años' };

  // meses completos
  const dy2 = new Date(dobMs);
  const ty2 = new Date(todayMs);
  let months = (ty2.getUTCFullYear() - dy2.getUTCFullYear()) * 12 + (ty2.getUTCMonth() - dy2.getUTCMonth());
  if (ty2.getUTCDate() < dy2.getUTCDate()) months -= 1;
  if (months >= 1) return { value: months, unit: 'meses', display: months + ' meses' };

  // días
  const days = Math.floor((todayMs - dobMs) / (24 * 60 * 60 * 1000));
  return { value: days, unit: 'días', display: days + ' días' };
}

export function formatDobForDocs(dob) {
  const m = DOB_RE.exec(typeof dob === 'string' ? dob : '');
  if (!m) return '';
  return m[3] + '/' + m[2] + '/' + m[1];
}
```

```js
// Listado de Problemas — manipulación inmutable.

const SECCIONES = ['activos', 'inactivos'];

function nuevoId() {
  return Date.now().toString(36) + Math.random().toString(36).slice(2, 8);
}

export function emptyListado(fecha, hora) {
  return {
    fecha: String(fecha || ''),
    hora: String(hora || ''),
    activos: [],
    inactivos: []
  };
}

function ensureSeccion(seccion) {
  if (!SECCIONES.includes(seccion)) {
    throw new Error('sección inválida: ' + seccion);
  }
}

export function addProblema(listado, seccion, datos) {
  ensureSeccion(seccion);
  const item = {
    id: nuevoId(),
    fecha: String((datos && datos.fecha) || ''),
    descripcion: String((datos && datos.descripcion) || '')
  };
  return Object.assign({}, listado, {
    [seccion]: (listado[seccion] || []).concat([item])
  });
}

export function removeProblema(listado, seccion, id) {
  ensureSeccion(seccion);
  const arr = listado[seccion] || [];
  const filtered = arr.filter(p => p.id !== id);
  if (filtered.length === arr.length) return listado;
  return Object.assign({}, listado, { [seccion]: filtered });
}

export function reorderProblema(listado, seccion, fromIndex, toIndex) {
  ensureSeccion(seccion);
  const arr = (listado[seccion] || []).slice();
  if (fromIndex < 0 || fromIndex >= arr.length) return listado;
  if (toIndex < 0 || toIndex >= arr.length) return listado;
  const [moved] = arr.splice(fromIndex, 1);
  arr.splice(toIndex, 0, moved);
  return Object.assign({}, listado, { [seccion]: arr });
}
```

```js
function ensureListadoForActive() {
  if (!activeId) return null;
  if (!listadoProblemas[activeId]) {

    var fecha = String(today.getDate()).padStart(2,'0')+'/'+String(today.getMonth()+1).padStart(2,'0')+'/'+today.getFullYear();

function renderListadoForm() {
  var c = document.getElementById('listado-form');
  if (!c || !activeId) { if (c) c.innerHTML = ''; return; }
  var lst = ensureListadoForActive();

  function renderRow(seccion, p) {
    return '<div class="listado-row" draggable="true" data-id="'+p.id+'" data-seccion="'+seccion+'">' +
```

```js
function setupListadoDragDrop() {
  var rows = document.querySelectorAll('.listado-row');
  rows.forEach(function(row) {
    row.addEventListener('dragstart', function(e) {
      row.classList.add('dragging');
      e.dataTransfer.setData('text/plain', JSON.stringify({
        id: row.dataset.id, seccion: row.dataset.seccion
      }));
    });
    row.addEventListener('dragend', function() { row.classList.remove('dragging'); });
    row.addEventListener('dragover', function(e) { e.preventDefault(); row.classList.add('drag-over'); });
    row.addEventListener('dragleave', function() { row.classList.remove('drag-over'); });
    row.addEventListener('drop', function(e) {
      e.preventDefault();
      row.classList.remove('drag-over');
      var data;
      try { data = JSON.parse(e.dataTransfer.getData('text/plain')); } catch (_) { return; }
      if (!data || data.seccion !== row.dataset.seccion) return; // solo dentro de misma sección
      var lst = ensureListadoForActive(); if (!lst) return;
      var arr = lst[data.seccion] || [];
      var fromIdx = arr.findIndex(function(p){ return p.id === data.id; });
      var toIdx = arr.findIndex(function(p){ return p.id === row.dataset.id; });
      if (fromIdx === -1 || toIdx === -1) return;
      listadoProblemas[activeId] = reorderProblema(lst, data.seccion, fromIdx, toIdx);
      saveState();
      renderListadoForm();
    });
  });
}
```

```python
#!/usr/bin/env python3
"""
Generates a DOCX 'Listado de Problemas' filling template_listado.docx.
Reads JSON from stdin, writes DOCX bytes to stdout.

JSON shape:
{
  "patient": { "nombre", "registro", "edad", "fechaNacimiento", "sexo", "area", "servicio", "cuarto", "cama" },
  "listado": { "fecha", "hora", "activos": [{fecha,descripcion}], "inactivos": [{fecha,descripcion}] },
  "doctor": { "doctorName", "grado" }
}
"""
import sys, json, zipfile, io, re, os, copy
from xml.etree import ElementTree as ET

TEMPLATE_PATH = os.environ.get(
    'TEMPLATE_PATH',
    os.path.join(os.path.dirname(os.path.abspath(__file__)), 'template_listado.docx')
)

W_NS = 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'
ET.register_namespace('w', W_NS)
NS = {'w': W_NS}

def esc(text):
    if not text:
        return ''
    return (str(text)
        .replace('&', '&amp;')
        .replace('<', '&lt;')
        .replace('>', '&gt;'))

def replace_t(xml, old_val, new_val):
    e_old = esc(old_val); e_new = esc(new_val)
    xml = xml.replace(f'<w:t>{e_old}</w:t>', f'<w:t>{e_new}</w:t>')
    xml = xml.replace(f'<w:t xml:space="preserve">{e_old}</w:t>',
                      f'<w:t xml:space="preserve">{e_new}</w:t>')
    return xml

def main():
    data = json.load(sys.stdin)
    patient = data.get('patient', {})
    listado = data.get('listado', {})
    doctor  = data.get('doctor', {})

    with zipfile.ZipFile(TEMPLATE_PATH, 'r') as zin:
        files = {n: zin.read(n) for n in zin.namelist()}

    xml = files['word/document.xml'].decode('utf-8')

    # ── Header replacements (appear twice in template) ───────────────────────
    nombre   = (patient.get('nombre') or '').upper()
    registro = patient.get('registro') or ''
    edad     = str(patient.get('edad') or '')
    dob      = patient.get('fechaNacimiento') or ''
    if dob and re.match(r'^\d{4}-\d{2}-\d{2}$', dob):
        y, m, d = dob.split('-')
        dob_disp = f'{d}/{m}/{y}'
    else:
        dob_disp = ''
    sexo     = patient.get('sexo') or ''
    area     = (patient.get('area') or '').upper()
    servicio = (patient.get('servicio') or '').upper()
    cuarto   = patient.get('cuarto') or ''
    cama     = patient.get('cama') or ''

    # Replace literal "MEDICINA INTERNA" del template con el servicio real (después
    # de header inserts para no chocar con possible cuerpo).
    xml = xml.replace('<w:t>MEDICINA INTERNA</w:t>', f'<w:t>{esc(servicio or "MEDICINA INTERNA")}</w:t>')

    # Insert values in fields. The template has empty placeholders between
    # field labels. We append spans inside the cells. For 3.0, simple approach:
    # find the "<w:t>NOMBRE</w:t>" runs (twice) and insert a sibling run with the value.

    # Heuristic: target the first <w:t xml:space="preserve"> </w:t> that follows each label.
    # A safer approach is to operate on parsed XML tree.
    root = ET.fromstring(xml)
    body = root.find('w:body', NS)

    def fill_after_label(label, value):
        """Find <w:t> elements equal to `label` (or containing `label:`) and set the
        following empty <w:t xml:space="preserve"> sibling to value."""
        nonlocal body
        if value is None: value = ''
        for t in body.iter(f'{{{W_NS}}}t'):
            if t.text and t.text.strip().rstrip(':').upper() == label.rstrip(':').upper():
                # find next <w:t> empty/whitespace-only
                # walk parent run, then next runs
                parent = None
                for r in body.iter(f'{{{W_NS}}}r'):
                    if t in list(r):
                        parent = r; break
                if parent is None: continue
                # find sibling runs in same paragraph
                # iterate through paragraph children
                for p in body.iter(f'{{{W_NS}}}p'):
                    runs = list(p.iter(f'{{{W_NS}}}r'))
                    if parent in runs:
                        idx = runs.index(parent)
                        for r2 in runs[idx+1:]:
                            for t2 in r2.iter(f'{{{W_NS}}}t'):
                                if not (t2.text and t2.text.strip()):
                                    t2.text = value
                                    return
                        return

    # Fill each field (header repeats twice → call twice)
    for _ in range(2):
        fill_after_label('NOMBRE', nombre)
        fill_after_label('REGISTRO', registro)
        fill_after_label('EDAD', edad)
        fill_after_label('FECHA DE NACIMIENTO', dob_disp)
        fill_after_label('SEXO', sexo)
        fill_after_label('CUARTO', cuarto)
        fill_after_label('CAMA', cama)

    # ── Body table: clone template row N times for activos and inactivos ─────
    # The template has rows numbered "1", "2", "3" with columns: FECHA, ACTIVOS, INACTIVOS.
    # Strategy: find the first <w:tr> after "FECHA / ACTIVOS / INACTIVOS" header that has
    # a numeric "1" cell; collect the row + the next two; remove them; clone first row N times,
    # filling columns based on activos (col index 1) and inactivos (col index 2).
    # If activos.len > inactivos.len, leftover activos rows leave inactivos blank, and vice-versa.

    # Find table that contains "ACTIVOS" header
    target_tbl = None
    for tbl in body.iter(f'{{{W_NS}}}tbl'):
        text_blob = ''.join((t.text or '') for t in tbl.iter(f'{{{W_NS}}}t'))
        if 'ACTIVOS' in text_blob and 'INACTIVOS' in text_blob:
            target_tbl = tbl; break

    if target_tbl is not None:
        rows = list(target_tbl.findall('w:tr', NS))
        # First row is header (FECHA/ACTIVOS/INACTIVOS); skip it.
        # Numbered rows follow.
        header_idx = 0
        for i, r in enumerate(rows):
            blob = ''.join((t.text or '') for t in r.iter(f'{{{W_NS}}}t'))
            if 'FECHA' in blob and 'ACTIVOS' in blob:
                header_idx = i; break
        data_rows = rows[header_idx+1:]
        # Use the first data row as the row template.
        if data_rows:
            row_template = copy.deepcopy(data_rows[0])
            # remove all existing data rows
            for r in data_rows:
                target_tbl.remove(r)

            activos = listado.get('activos') or []
            inactivos = listado.get('inactivos') or []
            n = max(len(activos), len(inactivos))

            for i in range(n):
                new_row = copy.deepcopy(row_template)
                cells = list(new_row.findall('w:tc', NS))
                # Cell 0: número (1, 2, 3, ...) or fecha — depende del template
                # Cell 1: activos[i].descripcion
                # Cell 2: inactivos[i].descripcion
                def set_cell_text(tc, text):
                    # clear existing text in cell, set first <w:t> to text
                    ts = list(tc.iter(f'{{{W_NS}}}t'))
                    for j, t in enumerate(ts):
                        t.text = text if j == 0 else ''
                # Cell 0: número de fila + fecha del activo si existe
                fecha_act = activos[i]['fecha'] if i < len(activos) else ''
                fecha_inact = inactivos[i]['fecha'] if i < len(inactivos) else ''
                col0 = str(i+1) + ('  ' + fecha_act if fecha_act else '') + (('  /  ' + fecha_inact) if fecha_inact else '')
                if len(cells) >= 1: set_cell_text(cells[0], col0)
                if len(cells) >= 2: set_cell_text(cells[1], activos[i]['descripcion'] if i < len(activos) else '')
                if len(cells) >= 3: set_cell_text(cells[2], inactivos[i]['descripcion'] if i < len(inactivos) else '')
                target_tbl.append(new_row)

    # ── Firma del residente ──────────────────────────────────────────────────
    doc_name = (doctor.get('doctorName') or '').strip()
    grado = (doctor.get('grado') or '').strip()
    if doc_name:
        firma_full = (grado + ' ' + doc_name).strip().upper()
        for t in body.iter(f'{{{W_NS}}}t'):
            if t.text and 'JORGE ADRIÁN MONTEMAYOR' in t.text.upper():
                t.text = firma_full

    # Write back
    xml_out = ET.tostring(root, encoding='unicode')
    files['word/document.xml'] = xml_out.encode('utf-8')

    out = io.BytesIO()
    with zipfile.ZipFile(out, 'w', zipfile.ZIP_DEFLATED) as zout:
        for n, b in files.items():
            zout.writestr(n, b)

    sys.stdout.buffer.write(out.getvalue())

if __name__ == '__main__':
    main()
```
